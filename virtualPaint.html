      def getContours(img):
          contours, hierarchy = cv2.findContours(img, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_NONE)
          x, y, w, h = 0, 0, 0, 0
          for cnt in contours:
              area = cv2.contourArea(cnt)
              if area > 500:
                  # cv2.drawContours(imgResult, cnt, -1, (255, 0, 0), 3)
                  peri = cv2.arcLength(cnt, True)

          return x + w // 2, y
